CREATE TABLE RECOMENDED_VS_MEDIAN AS
SELECT 
  A.NODE_TITLE,
  A.SEMESTER_TYPE_ID AS RECOMENDED_SEMESTER,
  B.MEDIAN_SEMESTER
FROM (SELECT NODE_TITLE, NODE_ID, SEMESTER_TYPE_ID
  FROM ATHENS15.VIS_MODULE_NODE) AS A
LEFT JOIN (SELECT *
  FROM ATHENS15.VIS_NODE_SEM_MEDIAN) AS B
  ON A.NODE_ID = B.NODE_ID
GROUP BY A.NODE_TITLE;
